Algoritmos de Busca em JavaScript

Esta página apresenta três algoritmos de busca implementados em JavaScript: busca linear, busca binária recursiva e busca binária não recursiva. Cada seção inclui o código, uma explicação e um botão para executar um exemplo com o array [10, 20, 30, 40, 50] e alvo 30.

1. Algoritmo de Busca Linear

A busca linear percorre a lista sequencialmente até encontrar o elemento alvo. É simples, mas ineficiente para listas grandes (complexidade O(n)).

function buscaLinear(arr, alvo) {
    for (let i = 0; i < arr.length; i++) {
        if (arr[i] === alvo) {
            return i;  // Retorna o índice se encontrado
        }
    }
    return -1;  // Retorna -1 se não encontrado
}

2. Algoritmo de Busca Binária Recursiva

A busca binária recursiva divide a lista ao meio a cada chamada, assumindo que a lista está ordenada. Complexidade O(log n).

function buscaBinariaRecursiva(arr, alvo, esquerda, direita) {
    if (esquerda > direita) {
        return -1;  // Não encontrado
    }
    let meio = Math.floor((esquerda + direita) / 2);
    if (arr[meio] === alvo) {
        return meio;  // Encontrado
    } else if (arr[meio] > alvo) {
        return buscaBinariaRecursiva(arr, alvo, esquerda, meio - 1);  // Busca na metade esquerda
    } else {
        return buscaBinariaRecursiva(arr, alvo, meio + 1, direita);  // Busca na metade direita
    }
}

3. Algoritmo de Busca Binária Não Recursiva

A versão não recursiva usa um loop para dividir a lista, também assumindo ordenação. Complexidade O(log n), mais eficiente em termos de espaço.

function buscaBinariaIterativa(arr, alvo) {
    let esquerda = 0;
    let direita = arr.length - 1;
    while (esquerda <= direita) {
        let meio = Math.floor((esquerda + direita) / 2);
        if (arr[meio] === alvo) {
            return meio;  // Encontrado
        } else if (arr[meio] > alvo) {
            direita = meio - 1;  // Busca na metade esquerda
        } else {
            esquerda = meio + 1;  // Busca na metade direita
        }
    }
    return -1;  // Não encontrado
}
